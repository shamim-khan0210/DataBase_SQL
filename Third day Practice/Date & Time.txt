//Date & Time


//RETURN CURRENT DATE
SELECT CURDATE();

//RETEUN CURRENT TIME
SELECT CURTIME();

//NOW FUNCTION
SELECT NOW();

//DATE & ADD DAYS ON DATE
SELECT ADDDATE('2023-8-24', INTERVAL 5 DAY);// Add 5 days on this date--2023-08-29

//DATE & SUBSTRACT DAYS ON DATE
SELECT SUBDATE('2023-8-24', INTERVAL 5 DAY);// Substract 5 days on this date--2023-08-19

//MAKE DATE 
SELECT MAKEDATE(2023, 277); // 2023-10-04

//MAKE DAY NAME
SELECT DAYNAME('2023-08-24'); //Thursday
